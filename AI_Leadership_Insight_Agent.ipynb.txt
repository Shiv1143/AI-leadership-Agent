    "import io\n",
    "import textwrap\n",
    "from pathlib import Path\n",
    "\n",
    "import pypdf\n",
    "import docx as python_docx\n",
    "\n",
    "\n",
    "# ── Document loaders ────────────────────────────────────────────────────────\n",
    "\n",
    "def load_pdf(path: str) -> str:\n",
    "    \"\"\"Extract text from a PDF file.\"\"\"\n",
    "    reader = pypdf.PdfReader(path)\n",
    "    return \"\\n\".join(page.extract_text() or \"\" for page in reader.pages)\n",
    "\n",
    "\n",
    "def load_docx(path: str) -> str:\n",
    "    \"\"\"Extract text from a .docx file.\"\"\"\n",
    "    doc = python_docx.Document(path)\n",
    "    return \"\\n\".join(p.text for p in doc.paragraphs)\n",
    "\n",
    "\n",
    "def load_txt(path: str) -> str:\n",
    "    \"\"\"Read a plain-text file.\"\"\"\n",
    "    return Path(path).read_text(encoding=\"utf-8\", errors=\"ignore\")\n",
    "\n",
    "\n",
    "def load_document(path: str) -> dict:\n",
    "    \"\"\"Auto-detect file type and return {source, content}.\"\"\"\n",
    "    ext = Path(path).suffix.lower()\n",
    "    loaders = {\".pdf\": load_pdf, \".docx\": load_docx, \".txt\": load_txt}\n",
    "    if ext not in loaders:\n",
    "        raise ValueError(f\"Unsupported file type: {ext}\")\n",
    "    content = loaders[ext](path)\n",
    "    return {\"source\": Path(path).name, \"content\": content}\n",
    "\n",
    "\n",
    "def load_documents_from_folder(folder: str) -> list[dict]:\n",
    "    \"\"\"Load all supported documents from a folder.\"\"\"\n",
    "    docs = []\n",
    "    for ext in (\"*.pdf\", \"*.docx\", \"*.txt\"):\n",
    "        for fp in Path(folder).glob(ext):\n",
    "            doc = load_document(str(fp))\n",
    "            docs.append(doc)\n",
    "            print(f\"  📄 Loaded: {doc['source']}  ({len(doc['content'])} chars)\")\n",
    "    return docs\n",
    "\n",
    "\n",
    "print(\"✅ Document loaders defined.\")"

    "\n",
    "# ── 4a. Chunking ─────────────────────────────────────────────────────────────\n",
    "\n",
    "splitter = RecursiveCharacterTextSplitter(\n",
    "    chunk_size=CONFIG[\"chunk_size\"],\n",
    "    chunk_overlap=CONFIG[\"chunk_overlap\"],\n",

    "chunks = []  # list of {source, chunk_id, text}\n",

    "# ── 4b. Embedding ─────────────────────────────────────────────────────────────\n",
    "\n",
    "print(f\"Loading embedding model: {CONFIG['embedding_model']} ...\")\n",
    "embedder = SentenceTransformer(CONFIG[\"embedding_model\"])\n",

    "texts = [c[\"text\"] for c in chunks]\n",
    "print(f\"Encoding {len(texts)} chunks...\")\n",
    "embeddings = embedder.encode(texts, show_progress_bar=True, convert_to_numpy=True)\n",

    "print(f\"\\n✅ Embeddings shape: {embeddings.shape}\")\n",
    "\n",
    "\n",
    "# ── 4c. Build FAISS index ─────────────────────────────────────────────────────\n",
    "\n",
    "dim = embeddings.shape[1]\n",
    "index = faiss.IndexFlatL2(dim)           # L2 distance (equivalent to cosine on normalised vecs)\n",
    "faiss.normalize_L2(embeddings)           # Normalise → cosine similarity\n",
    "index.add(embeddings)\n",
    "\n",
    "print(f\"✅ FAISS index built with {index.ntotal} vectors (dim={dim}).\")"

    "## Step 5 — Retrieval & Generation"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "import anthropic\n",
    "\n",
    "client = anthropic.Anthropic(api_key=os.environ[\"ANTHROPIC_API_KEY\"])\n",
    "\n",
    "\n",
    "# ── Retriever ─────────────────────────────────────────────────────────────────\n",
    "\n",
    "def retrieve(query: str, top_k: int = CONFIG[\"top_k\"]) -> list[dict]:\n",
    "    \"\"\"Embed the query and return the top-K most relevant chunks.\"\"\"\n",
    "    q_emb = embedder.encode([query], convert_to_numpy=True)\n",
    "    faiss.normalize_L2(q_emb)\n",
    "    distances, indices = index.search(q_emb, top_k)\n",
    "    results = []\n",
    "    for dist, idx in zip(distances[0], indices[0]):\n",
    "        if idx == -1:\n",
    "            continue\n",
    "        chunk = chunks[idx].copy()\n",
    "        chunk[\"similarity\"] = float(1 - dist / 2)  # convert L2 distance → cosine similarity\n",
    "        results.append(chunk)\n",
    "    return results\n",
    "\n",
    "\n",
    "# ── Prompt builder ────────────────────────────────────────────────────────────\n",
    "\n",
    "SYSTEM_PROMPT = \"\"\"You are a senior business analyst serving company leadership.\n",
    "You answer questions ONLY using the context excerpts provided — do not add outside knowledge.\n",
    "If the context does not contain enough information to answer, say so clearly.\n",
    "\n",
    "Guidelines:\n",
    "- Be concise but complete. Use bullet points for lists of items.\n",
    "- Cite the source document for each key fact (e.g. [Q3_2024_Earnings.txt]).\n",
    "- If numbers are mentioned, be precise — do not round unless necessary.\n",
    "- Highlight risks, red flags, or underperformance prominently.\n",
    "\"\"\"\n",
    "\n",
    "\n",
    "def build_prompt(question: str, context_chunks: list[dict]) -> str:\n",
    "    context_text = \"\\n\\n\".join(\n",
    "        f\"[Source: {c['source']}]\\n{c['text']}\"\n",
    "        for c in context_chunks\n",
    "    )\n",
    "    return f\"\"\"Use the following excerpts from company documents to answer the leadership question.\n",
    "\n",
    "=== CONTEXT EXCERPTS ===\n",
    "{context_text}\n",
    "\n",
    "=== LEADERSHIP QUESTION ===\n",
    "{question}\n",
    "\n",
    "Provide a clear, evidence-based answer citing the relevant documents.\"\"\"\n",
    "\n",
    "\n",
    "# ── Main agent function ───────────────────────────────────────────────────────\n",
    "\n",
    "def ask_agent(question: str, verbose: bool = False) -> str:\n",
    "    \"\"\"\n",
    "    Ask the leadership insight agent a natural-language question.\n",
    "    Returns a grounded answer citing source documents.\n",
    "    \"\"\"\n",
    "    # 1. Retrieve relevant chunks\n",
    "    context = retrieve(question)\n",
    "\n",
    "    if verbose:\n",
    "        print(f\"\\n🔍 Retrieved {len(context)} chunks:\")\n",
    "        for c in context:\n",
    "            print(f\"   [{c['similarity']:.2f}] {c['source']} | {c['text'][:80]}...\")\n",
    "        print()\n",
    "\n",
    "    # 2. Build prompt\n",
    "    user_prompt = build_prompt(question, context)\n",
    "\n",
    "    # 3. Call Claude\n",
    "    response = client.messages.create(\n",
    "        model=CONFIG[\"model\"],\n",
    "        max_tokens=CONFIG[\"max_tokens\"],\n",
    "        system=SYSTEM_PROMPT,\n",
    "        messages=[{\"role\": \"user\", \"content\": user_prompt}],\n",
    "    )\n",
    "\n",
    "    return response.content[0].text\n",
    "\n",
    "\n",
    "print(\"✅ Agent is ready. Use ask_agent('your question') to query.\")"
   ]

    "---\n",
    "\n",
    "## Architecture Summary\n",
    "\n",
    "| Component | Choice | Rationale |\n",
    "|---|---|---|\n",
    "| **LLM** | Claude claude-sonnet-4-20250514 | Strong instruction-following, accurate citations |\n",
    "| **Embeddings** | all-MiniLM-L6-v2 | Fast, free, runs on CPU, 384-dim vectors |\n",
    "| **Vector store** | FAISS (IndexFlatL2 + L2 normalisation) | In-memory, no server needed, exact cosine search |\n",
    "| **Chunking** | RecursiveCharacterTextSplitter (800 chars, 150 overlap) | Preserves sentence boundaries; overlap reduces context loss |\n",
    "| **File formats** | PDF · DOCX · TXT | Covers all common company document types |\n",
    "| **UI** | Gradio chat | Zero-config interactive interface |\n",
    "\n",
    "## Extending This Agent\n",
    "\n",
    "- **Larger documents / more files** → swap FAISS for [ChromaDB](https://www.trychroma.com/) or [Pinecone](https://www.pinecone.io/) for persistent storage  \n",
    "- **Better embeddings** → use `text-embedding-3-large` (OpenAI) or `voyage-2` (Anthropic) for higher accuracy  \n",
    "- **Hybrid search** → combine dense embeddings with BM25 keyword search (via `rank_bm25`)  \n",
    "- **Multi-turn conversation** → maintain message history and pass previous Q&A into Claude's context  \n",
    "- **Source highlighting** → return exact page numbers / paragraph offsets from PDF/DOCX  \n",
    "- **Task 2 – Autonomous agent** → add tool use (web search, database queries, code execution) to evolve into a decision-making system  \n"
